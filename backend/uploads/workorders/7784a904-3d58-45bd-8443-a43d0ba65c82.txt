# Dockerfile for playit.gg tunnel control service
# This image includes Python, your control scripts, and the playit.gg client

FROM python:3.11-slim

# Set workdir
WORKDIR /app

# Copy requirements and install dependencies
COPY requirements.txt ./
RUN pip install --no-cache-dir -r requirements.txt

# Download playit.gg client (Linux x86_64 example)
ADD https://github.com/playit-cloud/playit-agent/releases/download/v0.17.1/playit-linux-amd64 ./playit
RUN chmod +x ./playit

# Entrypoint: run the playit tunnel persistently
ENTRYPOINT ["./playit"]
